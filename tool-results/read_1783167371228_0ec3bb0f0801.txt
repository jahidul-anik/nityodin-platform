   334→-- Radius search query example (find plumbers within 10km)
   335→SELECT id, business_name, rating, 
   336→       ST_Distance(location, ST_MakePoint(90.4125, 23.8103)::geography) AS distance_meters
   337→FROM service_providers 
   338→WHERE category = 'plumbing'
   339→  AND is_available = true
   340→  AND ST_DWithin(
   341→      location, 
   342→      ST_MakePoint(90.4125, 23.8103)::geography,  -- User's location (Dhaka)
   343→      10000  -- 10km radius in meters
   344→  )

   346→LIMIT 20;
