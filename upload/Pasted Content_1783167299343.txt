CREATE EXTENSION IF NOT EXISTS postgis;

-- Merchants/Service Providers table with location
CREATE TABLE service_providers (
    id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    business_name VARCHAR(255),
    category VARCHAR(100),
    location GEOGRAPHY(POINT, 4326),  -- SRID 4326 for GPS coordinates
    service_radius_km INTEGER DEFAULT 50,
    rating DECIMAL(3,2),
    is_available BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Create spatial index for fast radius queries
CREATE INDEX idx_sp_location ON service_providers USING GIST (location);

-- Radius search query example (find plumbers within 10km)
SELECT id, business_name, rating, 
       ST_Distance(location, ST_MakePoint(90.4125, 23.8103)::geography) AS distance_meters
FROM service_providers 
WHERE category = 'plumbing'
  AND is_available = true
  AND ST_DWithin(
      location, 
      ST_MakePoint(90.4125, 23.8103)::geography,  -- User's location (Dhaka)
      10000  -- 10km radius in meters
  )
ORDER BY distance_meters
LIMIT 20;

    order_id UUID,

    disputed_by UUID REFERENCES users(id),
    disputed_at TIMESTAMP WITH TIME ZONE,
    resolved_by UUID REFERENCES users(id),  -- Admin who resolved

    Port            int           `mapstructure:"port"`

func (p PostgresConfig) ConnectionURL() string {
    return fmt.Sprintf(
        "postgresql://%s:%s@%s:%d/%s?sslmode=%s",
        p.User, p.Password, p.Host, p.Port, p.DBName, p.SSLMode,
    )
}

type RedisConfig struct {
    Host     string        `mapstructure:"host"`
    Port     int           `mapstructure:"port"`
    Password string        `mapstructure:"password"`
    DB       int           `mapstructure:"db"`
    PoolSize int           `mapstructure:"pool_size"`
    MinIdleConns int       `mapstructure:"min_idle_conns"`
}

func (r RedisConfig) ConnectionURL() string {
    password := ""
    if r.Password != "" {
        password = fmt.Sprintf(":%s@", r.Password)
    }
    return fmt.Sprintf("redis://%s%s:%d/%d",
        password, r.Host, r.Port, r.DB)
}

// IsExpired checks if session has expired

    return time.Now().After(s.ExpiresAt)
}

    return s.IsValid && !s.IsExpired()
}

    query := `
        UPDATE users SET
            password_hash = $2,
            updated_at = NOW(),

    return nil
}

// @Failure 401 {object} util.Response
// @Router /auth/refresh [post]
func (h *AuthHandler) RefreshToken(w http.ResponseWriter, r *http.Request) {
    // Extract refresh token from Authorization header
    // Note: Client should send refresh token in body or header (depending on your security preference)
    // Here we're using custom header for clarity: X-Refresh-Token
    refreshToken := r.Header.Get("X-Refresh-Token")
    if refreshToken == "" {
        // Fallback: try body
        var body struct {
            RefreshToken string `json:"refresh_token"`
        }
        if err := json.NewDecoder(r.Body).Decode(&body); err != nil || body.RefreshToken == "" {
            util.WriteError(w, errors.NewUnauthorized("Refresh token required"))
            return
        }
        refreshToken = body.RefreshToken
    }
    
    tokens, err := h.authService.RefreshToken(r.Context(), refreshToken)
    if err != nil {
        util.WriteError(w, err)
        return
    }
    
    response := dto.TokenResponse{
        AccessToken:  tokens.AccessToken,
        RefreshToken: tokens.RefreshToken,
        ExpiresIn:    tokens.ExpiresIn,
        TokenType:    tokens.TokenType,
    }
    
    util.WriteJSON(w, http.StatusOK, response)
}

// Logout handles POST /v1/auth/logout
// @Summary Logout user
// @Description Invalidates current session (requires authentication)
// @Tags Authentication
// @Security BearerAuth
// @Success 204 "No content"
// @Router /auth/logout [post]
func (h *AuthHandler) Logout(w http.ResponseWriter, r *http.Request) {
    // Extract session ID from token claims (set by auth middleware)
    sessionID, _ := r.Context().Value("session_id").(uuid.UUID)
    
    if err := h.authService.Logout(r.Context(), sessionID); err != nil {
        util.WriteError(w, err)
        return
    }
    
    w.WriteHeader(http.StatusNoContent)
}

// Context keys for storing authenticated user info in request context
type contextKey string

const (
    UserIDKey    contextKey = "user_id"
    SessionIDKey contextKey = "session_id"
    TokenKey    contextKey = "token_info"
)

// Authenticate creates middleware that validates JWT tokens
func Authenticate(validateFunc func(token string) (*util.TokenClaims, error)) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // 1. Extract Authorization header
            authHeader := r.Header.Get("Authorization")
            if authHeader == "" {
                util.WriteError(w, errors.NewUnauthorized("Authorization header required"))
                return
            }

            // 2. Parse "Bearer <token>" format
            parts := strings.SplitN(authHeader, " ", 2)
            if len(parts) != 2 || !strings.EqualFold(parts[0], "bearer") {
                util.WriteError(w, errors.NewUnauthorized("Invalid authorization header format. Expected: Bearer <token>"))
                return
            }

            tokenString := parts[1]

            // 3. Validate token
            claims, err := validateFunc(tokenString)
            if err != nil {
                util.WriteError(w, err) // Already wrapped as proper error
                return
            }

            // 4. Inject user info into request context
            ctx := context.WithValue(r.Context(), UserIDKey, claims.UserID)
            ctx = context.WithValue(ctx, SessionIDKey, claims.SessionID)
            ctx = context.WithValue(ctx, TokenKey, claims)

            // 5. Call next handler with enriched context
            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}

// HashPassword securely hashes a password using bcrypt
func HashPassword(password string, cost int) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), cost)
    if err != nil {
        return "", err
    }
    return string(bytes), nil
}

// VerifyPassword checks if a password matches a hash
func VerifyPassword(password, hash string) error {
    return bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))

type TokenClaims struct {
    UserID        uuid.UUID          `json:"user_id"`
    SessionID     uuid.UUID          `json:"session_id"`
    ActiveRole    *model.UserRoleType `json:"active_role,omitempty"`
    IsRoleVerified bool              `json:"is_role_verified"`
    jwt.RegisteredClaims
}

// GenerateAccessToken creates a short-lived JWT access token
func GenerateAccessToken(
    userID, sessionID uuid.UUID,
    secret string,
    ttl time.Duration,
) (string, int64, error) {
    now := time.Now()
    expiry := now.Add(ttl)
    
    claims := TokenClaims{
        UserID:    userID,
        SessionID: sessionID,
        RegisteredClaims: jwt.RegisteredClaims{
            Issuer:    "nityodin-identity-service",
            Subject:   userID.String(),
            Audience:  jwt.ClaimStrings{"nityodin-platform"},
            IssuedAt:  jwt.NewNumericDate(now),
            ExpiresAt: jwt.NewNumericDate(expiry),
            NotBefore: jwt.NewNumericDate(now),
            ID:        uuid.New().String(), // Unique token identifier (JTI)
        },
    }
    
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    signedToken, err := token.SignedString([]byte(secret))
    if err != nil {
        return "", 0, fmt.Errorf("failed to sign access token: %w", err)
    }
    
    return signedToken, int64(ttl.Seconds()), nil
}

RETURNS void AS $$ DECLARE
    partition_name text;
BEGIN
    partition_name := table_name || '_y' || to_char(start_date, 'YYYY') || 'm' || lpad(to_char(EXTRACT(MONTH FROM start_date)::int), 2, '0');
    
    EXECUTE format(
        'CREATE TABLE IF NOT EXISTS %I PARTITION OF %I FOR VALUES FROM (%L) TO (%L)',
        partition_name,
        table_name,
        start_date,
        end_date
    );
END;

Definition: Complete service outage affecting ALL users or critical security breach.

Examples:

All users unable to login (Identity Service down)
Payment processing completely broken (cannot complete transactions)
Data breach confirmed (user data exposed)
Database corruption or data loss risk
Complete platform unavailability (>90% error rate)
Response SLAs:

Acknowledgment: ≤ 5 minutes
Initial Triage: ≤ 10 minutes
First Update (Internal): ≤ 15 minutes
First Update (Public): ≤ 30 minutes (via Status Page)
Resolution Target: ≤ 1 hour (or workaround in place)
War Room: Required (all hands on deck)
Escalation: Immediate to CTO + VP Engineering

SEV2 - HIGH (🟠)
Definition: Significant degradation affecting LARGE user subset or critical feature broken.

Examples:

Location-based search failing (core discovery feature down)
Wallet balance showing incorrect amounts
OTP delivery failure (users cannot register/login)
Single region or availability zone outage
30-50% error rate on API endpoints
Payment delays >5 minutes
Response SLAs:

Acknowledgment: ≤ 15 minutes
Initial Triage: ≤ 30 minutes
First Update (Internal): ≤ 45 minutes
First Update (Public): ≤ 1 hour
Resolution Target: ≤ 4 hours
War Room: Recommended for complex issues
Escalation: Within 30 minutes to Engineering Manager

SEV3 - MEDIUM (🟡)
Definition: Partial degradation or non-critical feature broken; workaround available.

Examples:

Profile photo upload failing (can skip, retry later)
Push notifications delayed or not delivering
Admin dashboard slow/unresponsive
Single microservice degraded (not identity/wallet/location)
5-15% error rate on non-critical endpoints
Reporting/analytics features not working
Response SLAs:

Acknowledgment: ≤ 1 hour
Initial Triage: ≤ 2 hours
Resolution Target: ≤ 24 hours (next business day acceptable if low impact)
War Room: Not required
Escalation: To Team Lead within 4 hours if unresolved

SEV4 - LOW (🟢)
Definition: Minor issue with minimal user impact; cosmetic or edge case problems.

Examples:

Typos in error messages
Minor UI glitch on specific device/browser combination
Logging verbosity too high (cost impact, not functional)
Non-production environment issues
Feature request disguised as bug
Documentation errors
Response SLAs:

Acknowledgment: ≤ 1 business day
Resolution Target: ≤ 2 sprints (or backlog prioritization)
War Room: Never required
Escalation: Standard sprint planning process

# Standard Operating Procedure for Receiving an Alert

Step 1: ACKNOWLEDGE (0-2 minutes)
  Action: 
    - Open PagerDuty/Opsgenie alert
    - Click "Acknowledge" to stop paging
    - Note timestamp of acknowledgment
    
  Check:
    - Is this a legitimate alert or false positive?
    - Look at dashboard/graphs before taking action

Step 2: ASSESS (2-10 minutes)
  Actions:
    - Identify affected service(s) from alert metadata
    - Check monitoring dashboard (Datadog/Grafana):
      * Error rates (4xx, 5xx)
      * Latency percentiles (p50, p95, p99)
      * Throughput (requests/sec)
      * Infrastructure metrics (CPU, memory, disk, network)
    - Review recent deployments (any deploy in last 2 hours?)
    - Check #incidents Slack channel for related reports
    
  Output:
    - Determine severity (SEV1-4)
    - Create incident channel if SEV1/SEV2
    - Brief initial assessment message

Step 3: COMMUNICATE (Immediate for SEV1/SEV2)
  Internal (Slack #incidents):
    "🚨 SEV{X} Incident Declared
     Affected: {service_name}
     Impact: {description}
     Investigating: @{your_handle}
     Dashboard: {link_to_dashboard}
     Runbook: {link_to_runbook}"
     
  External (if SEV1):
    Update Statuspage.io with "Investigating" status

Step 4: MITIGATE (Using Runbooks)
  Priority Order:
    1. Restore service (rollback, restart, scale up)
    2. Contain impact (disable feature, rate limit)
    3. Identify root cause (while service is stable)
    4. Implement permanent fix (after incident resolved)

Step 5: RESOLVE
  Actions:
    - Verify service is healthy (check all monitors green)
    - Wait 15 minutes for stability confirmation
    - Resolve PagerDuty incident
    - Update Statuspage to "Resolved"
    - Post summary in #incidents channel

Step 6: DOCUMENT (Within 24 hours)
  - Create post-incident document
  - Schedule post-mortem meeting (within 48 hours for SEV1/2)

🟠 RUNBOOK 4.2: DATABASE PERFORMANCE DEGRADATION
Trigger: PostgreSQL latency >500ms OR connection timeouts OR slow query alerts

Symptoms:

API endpoints timing out (504 Gateway Timeout)
"Database connection exhausted" errors in logs
Queries that normally take <100ms now taking >5 seconds
Immediate Actions (0-10 minutes):

sql

-- CONNECT TO DATABASE (use read replica for diagnostics if possible)
-- psql $DATABASE_URL

-- 1. CHECK ACTIVE CONNECTIONS AND STATE
SELECT state, count(*) 
FROM pg_stat_activity 
WHERE datname = 'nityodin' 
GROUP BY state;
-- Look for: many "active" connections stuck (blocking queries?)

-- 2. FIND LONG-RUNNING QUERIES (Current)
SELECT pid, now() - pg_stat_activity.query_start AS duration, query, state
FROM pg_stat_activity 
WHERE (now() - pg_stat_activity.query_start) > interval '5 minutes'
ORDER BY duration DESC;

-- 3. CHECK FOR LOCK CONTENTION
SELECT blocked_locks.pid AS blocked_pid,
       blocking_locks.pid AS blocking_pid,
       blocked_activity.query AS blocked_statement,
       blocking_activity.query AS current_statement_in_blocking_process
FROM pg_catalog.pg_locks blocked_locks
JOIN pg_catalog.pg_locks blocking_locks 
  ON blocking_locks.locktype = blocked_locks.locktype
  AND blocking_locks.DATABASE IS NOT DISTINCT FROM blocked_locks.DATABASE
  AND blocking_locks.relation IS NOT DISTINCT FROM blocked_locks.relation
  AND blocking_locks.page IS NOT DISTINCT FROM blocked_locks.page
  AND blocking_locks.tuple IS NOT DISTINCT FROM blocked_locks.tuple
  AND blocking_locks.virtualxid IS NOT DISTINCT FROM blocked_locks.virtualxid
  AND blocking_locks.transactionid IS NOT DISTINCT FROM blocked_locks.transactionid
  AND blocking_locks.classid IS NOT DISTINCT FROM blocked_locks.classid
  AND blocking_locks.objid IS NOT DISTINCT FROM blocked_locks.objid
  AND blocking_locks.objsubid IS NOT DISTINCT FROM blocked_locks.objsubid
  AND blocking_locks.pid != blocked_locks.pid
JOIN pg_catalog.pg_stat_activity blocked_activity ON blocked_activity.pid = blocked_locks.pid
JOIN pg_catalog.pg_stat_activity blocking_activity ON blocking_activity.pid = blocking_locks.pid
WHERE NOT blocked_activity.waiting;

-- 4. CHECK REPLICATION LAG (If using replicas)
SELECT client_addr, state, sent_lsn, write_lsn, flush_lsn, replay_lsn,
       pg_wal_lsn_diff(write_lsn, replay_lsn) AS lag_bytes
FROM pg_stat_replication;

-- 5. CHECK TABLE SIZES (Identify bloated tables)
SELECT relname AS table_name,
       pg_size_pretty(pg_total_relation_size(relid)) AS total_size,
       pg_size_pretty(pg_relation_size(relid)) AS data_size,
       pg_size_pretty(pg_indexes_size(relid)) AS index_size
FROM pg_catalog.pg_statio_user_tables
ORDER BY pg_total_relation_size(relid) DESC
LIMIT 20;

-- 6. CHECK VACUUM/ANALYZE STATUS (Auto-vacuum running?)
SELECT relname, 
       last_vacuum, 
       last_autovacuum, 
       last_analyze, 
       last_autoanalyze,
       dead_tuple_count
FROM pg_stat_user_tables 
WHERE n_dead_tup > 1000
ORDER BY n_dead_tup DESC
LIMIT 10;
Diagnosis Decision Tree:

text

What does connection check show?
├── Too many IDLE connections (max_connections reached?)
│   → Application connection pool leaking
│   FIX: Increase max_connections OR fix app pool (PgBouncer recommended)
│
├── Many ACTIVE queries stuck for >5 min
│   ├── Lock contention detected?
│   │   → KILL the blocking transaction (carefully!)
│   │   SELECT pg_terminate_pid(blocking_pid);
│   │
│   └── Just slow queries?
│       → Missing indexes or statistics stale
│       FIX: Run ANALYZE on affected tables, EXPLAIN ANALYZE slow queries
│
├── Replication lag high (>1GB)?
│   → Replica overloaded or network issue
│   FIX: Redirect read traffic to primary temporarily, contact DBA
│
└── Autovacuum not keeping up (high dead tuples)?
    → Table bloat causing performance issues
    FIX: Schedule manual VACUUM FULL during maintenance window
Emergency Mitigation Commands:

bash

# IF DATABASE IS COMPLETELY UNRESPONSIVE:

# Option 1: Kill long-running sessions (use with caution!)
psql -c "
SELECT pg_terminate_pid(pid) 
FROM pg_stat_activity 
WHERE pid != pg_backend_pid()
  AND state = 'active'
  AND query_start < NOW() - interval '10 minutes';
"

# Option 2: Scale up RDS instance (immediate performance boost)
# AWS Console → RDS → Instances → Modify → Larger Instance Class

# Option 3: Enable Read Replica for Offloading (if not already)
# Redirect SELECT queries to replica (reduces load on primary by 50-70%)

# Option 4: Emergency Connection Pooling (if using direct connections)
# Deploy PgBouncer or switch to serverless RDS Proxy
Prevention Measures (Implement After Incident):

 Set up PgBouncer/RDS Proxy for connection pooling
 Configure slow query log (log_min_duration_statement = 1000)
 Set up automatic alerts for connection count >80% of max
 Schedule weekly VACUUM ANALYZE for high-write tables

Symptoms:

API responses slower than normal (fallback to DB queries)
"Redis connection refused" errors

Immediate Actions (0-5 minutes):

# 2. CHECK BASIC CONNECTIVITY

# 3. CHECK MEMORY USAGE (Critical!)

# Look at:
#   used_memory_human
#   used_memory_peak_human
#   used_memory_percentage (should be <70% ideally)
#   maxmemory (is it set? what's the limit?)

# 4. CHECK CONNECTION COUNT

# Compare against: redis-cli CONFIG GET maxclients

# 5. CHECK SLOW LOG (if enabled)

# Shows slowest operations

# 6. CHECK KEY SPACE (How many keys? Any patterns?)
redis-cli DBSIZE
redis-cli INFO keyspace
Common Scenarios & Fixes:

Scenario A: Redis Out of Memory (OOM)
Symptom: Redis crashed or evicting keys aggressively

bash

# CHECK: Did Redis hit memory limit?
redis-cli CONFIG GET maxmemory
redis-cli INFO memory | grep used_memory

# IF OOM:
# 1. Check eviction policy (should NOT be "noeviction" for cache)
redis-cli CONFIG GET maxmemory-policy
# Recommended: allkeys-lru (least recently used)

# 2. Emergency: Flush non-critical cache keys
# WARNING: This will cause cache misses but prevents crash
# Only do this if Redis is about to crash!
redis-cli --scan --pattern "session:*" | head -10000 | xargs redis-cli DEL
# Or flush entire cache (last resort):
# redis-cli FLUSHDB  # ⚠️ DESTRUCTIVE - Causes temporary slowdown

# 3. Better approach: Scale vertically (increase memory)
# ElastiCache → Modify → Increase node type (e.g., cache.t3.micro → cache.t3.medium)

# 4. Long-term: Add more shards (horizontal scaling)
Scenario B: Redis Master Failover (Cluster Mode)
Symptom: Write failures, connection resets

bash

# CHECK: Cluster topology
redis-cli -c CLUSTER NODES
# Look for: fail? status, master/slave assignments

# IF FAILOVER IN PROGRESS:
# 1. Wait 30 seconds for auto-failover to complete (Elasticache handles this)
# 2. Monitor: redis-cli -c CLUSTER INFO

# APPLICATION SIDE:
# Ensure your Redis client supports cluster mode (redis-go-cluster, ioredis)
# And has automatic reconnection logic

# TEMPORARY WORKAROUND:
# Point application to known-good node manually (update config/redeploy)
Scenario C: High Latency
Symptom: Redis responding but slowly (>100ms per command)

bash

# CHECK: Is Redis doing heavy lifting?
redis-cli --latency-history -i 1
# Runs for 1 minute, shows latency distribution

# LIKELY CAUSES:
# 1. BIG KEYS (keys with millions of elements)
redis-cli --bigkeys
# If found: Consider splitting large keys or using hashes

# 2. SLOW COMMANDS (KEYS *, FLUSHALL on production)
redis-cli SLOWLOG GET 50
# Find and eliminate slow commands in application code

# 3. NETWORK ISSUES (cross-AZ latency)
# Move Redis to same AZ as application servers
# Or enable cluster mode with local reads

# 4. CPU SATURATION (Redis is single-threaded!)
redis-cli INFO cpu
# If used_cpu_sys very high: Optimize commands or scale out (sharding)
Application-Side Fallback Behavior:

Your application should gracefully degrade when Redis is down:

go

// Example: Go service with Redis fallback
func GetSessionFromCache(ctx context.Context, sessionID string) (*Session, error) {
    // Try Redis first (fast path)
    session, err := redisClient.Get(ctx, fmt.Sprintf("session:%s", sessionID)).Result()
    if err == nil {
        return deserialize(session), nil
    }
    
    // Redis failed - fallback to database (slower but works)
    log.Warn("Redis cache miss, falling back to database",
        "session_id", sessionID,
        "error", err,
    )
    
    dbSession, err := sessionRepo.GetByID(ctx, sessionID)
    if err != nil {
        return nil, fmt.Errorf("both cache and DB failed: %w", err)
    }
    
    // Optionally: Repopulate cache asynchronously
    go func() {
        ctxBg := context.Background()
        if cacheErr := redisClient.Set(ctxBg, cacheKey, serialize(dbSession), 1*time.Hour).Err(); cacheErr != nil {
            log.Error("Failed to repopulate cache", "error", cacheErr)
        }
    }()
    
    return dbSession, nil
}

Scenario B: Balance Discrepancy
Cause: Race condition, failed write after debit, or caching issue

sql

-- Calculate expected balance vs actual balance
WITH user_ledger AS (
    SELECT wallet_id,
           sum(CASE WHEN direction = 'credit' THEN amount ELSE -amount END) as calculated_balance
    FROM transactions
    WHERE wallet_id = 'affected-wallet-uuid'
      AND transaction_status = 'completed'
    GROUP BY wallet_id
)
SELECT w.id, 
       w.available_balance as displayed_balance,
       l.calculated_balance as expected_balance,
       w.available_balance - l.calculated_balance as discrepancy
FROM wallets w
JOIN user_ledger l ON w.id = l.wallet_id
WHERE ABS(w.available_balance - l.calculated_balance) > 0.01;  -- More than 1 paisa difference

-- DO NOT AUTO-FIX. This requires manual audit.

# Option 2: External certificates (Let's Encrypt, DigiCert, etc.)
# Use cert-manager in Kubernetes (auto-renewal)

#!/bin/bash
# check_cert_expiry.sh - Run daily via cron
DOMAINS=("api.nityodin.com" "www.nityodin.com" "*.nityodin.com")
DAYS_THRESHOLD=30

for domain in "${DOMAINS[@]}"; do
  expiry_date=$(echo | openssl s_client -servername $domain -connect $domain:443 2>/dev/null | openssl x509 -noout -enddate 2>/dev/null | cut -d= -f2)
  expiry_epoch=$(date -d "$expiry_date" +%s)
  now_epoch=$(date +%s)
  days_until_expiry=$(( ($expiry_epoch - $now_epoch) / 86400 ))
  
  if [ $days_until_expiry -lt $DAYS_THRESHOLD ]; then
    echo "WARNING: Certificate for $domain expires in $days_until_expiry days ($expiry_date)"
    # Send alert to PagerDuty/Slack
    curl -X POST "$SLACK_WEBHOOK_URL" \
      -H 'Content-type: application/json' \
      --data "{\"text\":\"🔴 Certificate Expiring: $domain in $days_until_expiry days\"}"
  fi
done

🟠 RUNBOOK 4.10: HIGH CPU/MEMORY UTILIZATION
Trigger: CPU >80% sustained for >15 minutes OR Memory >90% with OOM kills

Symptoms:

Containers getting OOMKilled (Out of Memory)
High API latency (CPU starvation)
Pod throttling (CPU limits hit)
Node becoming NotReady (resource pressure)
Diagnosis:

bash

# 1. IDENTIFY HUNGRY PODS
kubectl top pods --sort-by=cpu
kubectl top pods --sort-by=memory

# Look for outliers consuming disproportionate resources

# 2. CHECK RESOURCE LIMITS VS REQUESTS
# Are pods requesting enough resources?
kubectl get pods -o custom-columns=\
'NAME:.metadata.name,\
CPU_REQ:.spec.containers[0].resources.requests.cpu,\
CPU_LIMIT:.spec.containers[0].resources.limits.cpu,\
MEM_REQ:.spec.containers[0].resources.requests.memory,\
MEM_LIMIT:.spec.containers[0].resources.limits.memory,\
CPU_USE:.status.containerStatuses[0].resources.usage.cpu,\
MEM_USE:.status.containerStatuses[0].resources.usage.memory'

# If USE接近LIMIT: About to be throttled/killed
# If REQUEST << LIMIT: Underprovisioned (scheduler may not place well)

# 3. CHECK NODE HEALTH
kubectl describe nodes | grep -A 10 "Conditions"
# Look for: MemoryPressure, PIDPressure, DiskPressure, NodeReady

# 4. INSPECT PROCESS INSIDE PROBLEMATIC POD
kubectl exec -it <pod-name> -- top -o %MEM | head -20
# Which process is eating memory?

kubectl exec -it <pod-name> -- sh -c "cat /proc/*/status 2>/dev/null | grep -A2 VmRSS | sort -k2 -nr | head -10"
# Go-specific: Check goroutine leaks
kubectl exec -it <pod-name> -- sh -c "wget -qO- localhost:6060/debug/pprof/goroutine?debug=1 | head -50"
# (Requires pprof endpoint enabled in Go binary)
Mitigation by Scenario:

Scenario A: Memory Leak (Gradual increase until OOM)
Likely Cause: Goroutine leak (Go), connection pool growing unbounded, cache growing unbounded

Fixes:

bash

# Immediate: Restart leaking pods (buys time)
kubectl rollout restart deployment/<service-name>

# Short-term: Increase memory limit (delays next OOM)
kubectl patch deployment <service-name> -p '{"spec":{"template":{"spec":{"containers":[{"name":"app","resources":{"limits":{"memory":"2Gi"}}}]}}}}'

# Long-term: Developer must find and fix leak
# Use profiling: kubectl port-forward <pod> 6060:6060
# Open: http://localhost:6060/debug/pprof/heap
# Compare heap profiles over time to identify leak
Scenario B: CPU Spike (Sudden, possibly traffic-related)
Likely Cause: Traffic surge, inefficient query, background job storm

Fixes:

bash

# Immediate: Scale horizontally (add more pods)
kubectl scale deployment <service-name> --replicas=10

# If autoscaler enabled, check why it didn't kick in:
kubectl describe hpa  # HorizontalPodAutoscaler
# Maybe min/max bounds too tight?

# Short-term: Reduce non-critical workloads
# Pause batch jobs, reduce background sync frequency
# Set feature flags to disable heavy features

# Check: Is there a specific endpoint causing load?
# Datadog APM → Services → <service-name> → Throughput by resource
# Identify hot endpoint, consider caching or rate limiting

// offline-sync-principles.ts

/**
 * Nityodin Offline-First Principles
 * 
 * 1. LOCAL IS SOURCE OF TRUTH (when offline)
 *    - App works fully without network
 *    - All writes go to local DB first
 *    - User sees their changes immediately
 * 
 * 2. OPTIMISTIC UI UPDATES
 *    - Assume success, show success immediately
 *    - Queue operation for later sync
 *    - Only show errors if sync fails permanently
 * 
 * 3. EVENTUAL CONSISTENCY
 *    - Conflicts resolved with CRDTs (Conflict-free Replicated Data Types)
 *    - Last-Write-Wins for simple fields
 *    - Custom merge functions for complex data (inventory quantities)
 * 
 * 4. GRACEFUL DEGRADATION
 *    - Show stale data with "last synced" timestamp
 *    - Disable features requiring real-time data (live prices?)
 *    - Queue operations intelligently (batch related ops)
 *

  private resolveProfileConflict(local: any, server: any): ConflictResolution<any> {
    const merged = { ...local };
    let hasConflicts = false;
    
    // Simple fields: LWW
    const lwwFields = ['fullName', 'bio', 'avatarUrl', 'preferredLanguage'];
    for (const field of lwwFields) {
      if (local[field] !== server[field]) {
        const localTime = new Date(local.updatedAt || 0).getTime();
        const serverTime = new Date(server.updatedAt || 0).getTime();
        merged[field] = localTime > serverTime ? local[field] : server[field];
        hasConflicts = true;
      }
    }
    
    // Arrays: Union (phones, emails, addresses)
    const arrayFields = ['phoneNumbers', 'emails', 'addresses'];
    for (const field of arrayFields) {
      if (JSON.stringify(local[field]) !== JSON.stringify(server[field])) {
        const combined = [...new Set([...(local[field] || []), ...(server[field] || []))];
        merged[field] = combined;
        hasConflicts = true;
      }
    }
    
    return {
      resolved: true,
      strategy: hasConflicts ? 'merge' : 'local_wins',
      mergedData: merged,
      newVersion: Math.max(local._version || 0, server._version || 0) + 1,
    };

// Utility: Format time until auto-release for display
function getTimeUntil(isoString: string): string {
  const now = new Date();
  const target = new Date(isoString);
  const diff = target.getTime() - now.getTime();
  
  if (diff <= 0) return 'Releasing soon...';
  
  const hours = Math.floor(diff / (1000 * 60 * 60));
  const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
  
  if (hours > 24) {
    const days = Math.floor(hours / 24);
    return `${days} days ${hours % 24}h remaining`;
  }
  
  if (hours > 0) {
    return `${hours}h ${minutes}m remaining`;
  }
  
  return `${minutes} minutes remaining`;
}

// EnsureAtomicity checks that concurrent operations on the same hold don't cause issues
func (s *EscrowSecurity) EnsureAtomicity(
    ctx context.Context,
    holdID uuid.UUID,
    expectedStatus model.EscrowStatus,
    operation string,
) error {
    
    // Use SELECT FOR UPDATE (pessimistic lock) or optimistic version checking
    current, err := s.escrowRepo.GetForUpdate(ctx, holdID)
    if err != nil {
        return err
    }
    
    if current.Status != expectedStatus {
        return errors.NewConflict(fmt.Sprintf(
            "Concurrency conflict: Expected status %s but found %s. Operation '%s' cannot proceed.",
            expectedStatus,
            current.Status,
            operation,
        ))
    }
    
    return nil
}
